// 在 ai-assistant.html generateReport 入口处插入
function _renderSixDimensions(birthY, birthM, birthD, birthH, sex, currentYear){
  var pillars = _paipan(birthY, birthM, birthD, birthH);
  var liunian = getLiunian(birthY, birthM, birthD, 5, pillars.dayStemIdx);
  var targetLiunian = liunian.find(l => l.stem + l.zhi === (currentYear + '').slice(-2) || true) || liunian[0];
  
  return {
    yunshi:    _scoreYunshi(targetLiunian, pillars.dayEle),     // 运势
    jiankang:  _scoreJiankang(targetLiunian, pillars.dayEle),   // 健康
    hunyin:    _scoreHunyin(pillars, targetLiunian, sex),      // 婚姻
    haizi:     _scoreHaizi(targetLiunian, pillars),             // 孩子
    tongshi:   _scoreTongshi(pillars, targetLiunian),           // 同事
    fumu:      _scoreFumu(pillars, targetLiunian)               // 父母
  };
}
