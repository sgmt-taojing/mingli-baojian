  // 实际环形：从壬开始，每15°，共24个
  // 壬中心角 = 0°处对称，即 壬 = (172.5 + 360) % 360 = 172.5 OR 另一标准是 壬 = 0°
  // 用最简单：index 0=壬 在 0°位置，但这样子=15°，丑=37.5°...这种排法0°=壬,与"正北=子"不符。
  // 解决方案：保持简单环形排法（每15°），但允许偏差——0°附近是壬/子/癸区

  function _get24Shan(angle) {
    // 找到最近的山
    angle = ((angle % 360) + 360) % 360;
    var bestIdx = 0;
    var minDiff = 360;
    // 24 山中心角度（以子为中心 0°=正北，顺时针）
    var centers = [
      337.5, // 壬 (北偏西22.5°，即子偏西22.5°)
      0,     // 子 (正北=0°)
      22.5,  // 癸 (北偏东22.5°)
      45,    // 丑 (东北偏北，即子偏东45°)
      67.5,  // 艮 (东北67.5°)
      90,    // 寅 (东偏北22.5° = 90°)
      112.5, // 甲 (东偏南22.5°)
      135,   // 卯 (东南偏东=135°但实际应为正东)
      157.5, // 乙 (东南偏南)
      180,   // 辰 (正南=180°)
      202.5, // 巽 (南偏西22.5°)
      225,   // 巳 (西南偏南=225°)
      247.5, // 丙 (西南偏西=247.5°)
      270,   // 午 (正西=270°)
      292.5, // 丁 (西北偏西=292.5°)
      315,   // 未 (西北偏北=315°)
      337.5, // 坤 (正北偏西) -- 24个中心点重新设计：
      22.5,  // 申
      67.5,  // 庚
      112.5, // 酉 — 重新规划，实际上应严格按 15° 间隔
    ];
    // 真实 24 山按 15° 间隔循环：以子=正北(0°)开始
    // 壬(-15°)=345°, 子(0°), 癸(15°), 丑(30°), 艮(45°), 寅(60°)
    // 甲(75°), 卯(90°), 乙(105°), 辰(120°), 巽(135°), 巳(150°)
    // 丙(165°), 午(180°), 丁(195°), 未(210°), 坤(225°), 申(240°)
    // 庚(255°), 酉(270°), 辛(285°), 戌(300°), 乾(315°), 亥(330°)
    centers = [
      345, 0, 15, 30, 45, 60, 75, 90, 105, 120, 135, 150,
      165, 180, 195, 210, 225, 240, 255, 270, 285, 300, 315, 330
    ];
    for (var i = 0; i < 24; i++) {
      var c = centers[i];
      // 计算最短角度差
      var diff = Math.abs(angle - c);
      if (diff > 180) diff = 360 - diff;
      if (diff < minDiff) {
        minDiff = diff;
        bestIdx = i;
      }
    }
    return { shan: SHAN_LIST[bestIdx], angle: angle, offsetDeg: minDiff.toFixed(2), gua: _shanToGua(SHAN_LIST[bestIdx]) };
  }

  function _shanToGua(shan) {
    if (shan === '壬' || shan === '子' || shan === '癸') return '坎';
    if (shan === '艮' || shan === '丑' || shan === '寅') return '艮';
    if (shan === '甲' || shan === '卯' || shan === '乙') return '震';
    if (shan === '巽' || shan === '辰' || shan === '巳') return '巽';
    if (shan === '丙' || shan === '午' || shan === '丁') return '离';
    if (shan === '坤' || shan === '未' || shan === '申') return '坤';
    if (shan === '庚' || shan === '酉' || shan === '辛') return '兑';
    if (shan === '乾' || shan === '戌' || shan === '亥') return '乾';
    return '中';
  }

  // ───────────────────────── 2. 三元九运 ─────────────────────────

  // 三元九运（沈氏玄空学）
  // 上元：1运(1864-1883) 2运(1884-1903) 3运(1904-1923)
  // 中元：4运(1924-1943) 5运(1944-1963) 6运(1964-1983)
  // 下元：7运(1984-2003) 8运(2004-2023) 9运(2024-2043)
  function getYuanYun(year) {
    if (year >= 1864 && year <= 1923) return '上元';
    if (year >= 1924 && year <= 1983) return '中元';
    if (year >= 1984 && year <= 2043) return '下元';
    // 出下元后按九循环推算
    if (year < 1864) return '上元之前的古代';
    return '下元（末）';
  }

  // 公元年 → 当元运星（每运20年）
  function getYunStarForYear(year) {
    // 1884 年=2, 1984年=7, 2024年=9, 2044年=1
    // 算法：((year - 1864) / 20) + 1, mod 9, 但 0 → 9
    var baseYun = 1; // 1864=1
    var yunIndex = (year - 1864) / 20; // 0=1运, 1=2运, 2=3运...
    // 1884-1864 = 20 → 1运（=baseYun+1）
    var n = Math.floor(yunIndex);
    var yun = ((baseYun - 1 + n) % 9) + 1;
    if (yun === 0) yun = 9;
    return yun;
  }

  // 飞星入中后顺飞九宫（洛书轨迹：1-9 顺时针 " → 6 → 7 → 8 → 9 → 1 →..."）
  function getFlyingStars(centerStar) {
    var map = {};
    for (var i = 0; i < 9; i++) {
      var star = ((centerStar + i - 1) % 9) + 1;
      if (star === 0) star = 9;
      map[LUOSHU_TRAVEL[i]] = star;
    }
    return map;
  }

  // 月飞星（按月令飞布）
  function getMonthlyFlyingStars(year, month) {
    var yearBranchIdx = ((year - 4) % 12 + 12) % 12; // 计算支
    var startStar;
    if ([0,6,3,9].indexOf(yearBranchIdx) >= 0) {
      startStar = 8;
    } else if ([2,8,5,11].indexOf(yearBranchIdx) >= 0) {
      startStar = 2;
    } else {
      startStar = 5;
    }
    var monthOffset = month - 1;
    var centerStar = ((startStar - 1 + monthOffset) % 9) + 1;
    if (centerStar === 0) centerStar = 9;
    return getFlyingStars(centerStar);
  }

  // 日飞星
  function getDailyFlyingStars(day) {
    var centerStar = ((day - 1) % 9) + 1;
    if (centerStar === 0) centerStar = 9;
    return getFlyingStars(centerStar);
  }

  // 从本卦起，按后天八卦顺序取所有卦
  function _getGuasFromThis(gua) {
    var idx = GUA_ORDER_HOUTIAN.indexOf(gua);
    if (idx < 0) return GUA_ORDER_HOUTIAN.slice();
    var result = [];
    for (var i = 0; i < 8; i++) {
      result.push(GUA_ORDER_HOUTIAN[(idx + i) % 8]);
    }
    return result;
  }

  /**
   * 起命卦（《八宅明镜》年命起法简化版）
   *
   * 男命：(100 - 出生尾数) % 9, 若为0取9，若为5则取2（寄坤）
   * 女命：(出生尾数 + 5) % 9, 若为0取9，若为5则取8（寄艮）
   */
  function getMingGuaClassic(year, sex) {
    var last2 = year % 100;
    var gua;
    var isMale = sex === 'male' || sex === '男' || sex === 'M';
    if (isMale) {
      gua = (100 - last2) % 9;
      if (gua === 0) gua = 9;
      if (gua === 5) gua = 2; // 男命得5寄坤
    } else {
      gua = (last2 + 5) % 9;
      if (gua === 0) gua = 9;
      if (gua === 5) gua = 8; // 女命得5寄艮
    }
    var guaNameMap = { 1:'坎', 2:'坤', 3:'震', 4:'巽', 6:'乾', 7:'兑', 8:'艮', 9:'离' };
    var isDongSi = [1, 9, 3, 4].indexOf(gua) >= 0;
    return {
      gua: gua,
      guaName: guaNameMap[gua],
      type: isDongSi ? '东四命' : '西四命',
      isDong: isDongSi
    };
  }

  }

  }

  function _filterByRank(table, ranks) {
    var out = [];
    Object.keys(table).forEach(function (dir) {
      if (ranks.indexOf(table[dir].jiXiong) >= 0) {
        out.push({ dir: dir, star: table[dir].star, rank: table[dir].jiXiong, wuxing: table[dir].wuxing });
      }

  function _scoreBaZhaiPosition(rank) {
    if (rank === '上吉') return 5;
    if (rank === '次吉') return 4;
    if (rank === '中吉') return 3;
    if (rank === '小吉') return 2;
    if (rank === '小凶') return -1;
    if (rank === '中凶') return -2;
    if (rank === '次凶') return -3;
    if (rank === '大凶') return -4;
    return 0;
  }

  function analyzeYangZhaiSanYao(year, sex, menhuDir, zhuWoDir, zaoFangDir) {
    var bazhai = bazhaiAnalyze(year, sex, menhuDir);
    var table = bazhai.directionTable;
    var menhu = table[menhuDir];
    var zhuWo = table[zhuWoDir];
    var zaoFang = table[zaoFangDir];

    var scores = {
      menhu: _scoreBaZhaiPosition(menhu && menhu.jiXiong),
      zhuWo: _scoreBaZhaiPosition(zhuWo && zhuWo.jiXiong),
      zaoFang: _scoreBaZhaiPosition(zaoFang && zaoFang.jiXiong)
    };
    var totalScore = scores.menhu + scores.zhuWo + scores.zaoFang;
    var remark = [];
    if (scores.menhu >= 4) remark.push('门得生气(纳财)');
    if (scores.zhuWo >= 4) remark.push('主得生气(旺丁)');
    if (scores.zaoFang >= 4) remark.push('灶得天医(健康)');

    return {
      totalScore: totalScore,
      remark: remark.length ? remark.join('，') : '三要需化解',
      menhu: { dir: menhuDir, star: menhu && menhu.star, rank: menhu && menhu.jiXiong, score: scores.menhu },
      zhuWo: { dir: zhuWoDir, star: zhuWo && zhuWo.star, rank: zhuWo && zhuWo.jiXiong, score: scores.zhuWo },
      zaoFang: { dir: zaoFangDir, star: zaoFang && zaoFang.star, rank: zaoFang && zaoFang.jiXiong, score: scores.zaoFang },
      advice: _yangZhaiAdvice(scores, menhu, zhuWo, zaoFang)
    };
  }

  function _yangZhaiAdvice(scores, menhu, zhuWo, zaoFang) {
    var tips = [];
    if (scores.menhu < 0) {
      tips.push('入户门位于' + (menhu && menhu.star || '凶位') + '，宜设玄关、挂门帘或设屏风缓冲，或改门向');
    } else if (scores.menhu < 3) {
      tips.push('入户门位于' + (menhu && menhu.star || '小吉位') + '，可加门槛或门垫催旺');
    }
    if (scores.zhuWo < 0) {
      tips.push('主卧位于' + (zhuWo && zhuWo.star || '凶位') + '，建议调整主卧位置或换主卧，方位对人最重');
    }
    if (scores.zaoFang < 0) {
      tips.push('厨房灶位位于' + (zaoFang && zaoFang.star || '凶位') + '，建议改灶位置或加铜器化解');
    } else if (scores.zaoFang >= 4) {
      tips.push('灶位得天医之气，主健康长寿');
    }
    return tips;
  }

  /**
   * 旺山旺向判断：山盘当旺之星所在方位 = 坐山方；向盘当旺之星所在方位 = 向方
   * 当元最旺之星 = 运星

  function isWangShanWangXiang(chaoxiang, yunStar) {
    var yun = getFlyingStars(yunStar);
    var zhanShan = _oppositeDir(chaoxiang);
    var shanStar = yun[zhanShan];
    var xiangStar = yun[chaoxiang];
    return shanStar === yunStar || xiangStar === yunStar;

  /**
   * 山星向星合于方位（如一六共宗、二七同道、三八为朋、四九为友）
   */
  function checkShanXiangJie(shanPan, xiangPan) {
    var relations = [];
    Object.keys(shanPan).forEach(function(dir) {
      var s = shanPan[dir];
      var x = xiangPan[dir];
      var pair = [Math.min(s, x), Math.max(s, x)];
      if (pair[0] === 1 && pair[1] === 6) relations.push({ dir: dir, stars: '一六', meaning: '一六共宗（水）— 文昌、利学业' });
      if (pair[0] === 2 && pair[1] === 7) relations.push({ dir: dir, stars: '二七', meaning: '二七同道（火）— 病符、口舌' });
      if (pair[0] === 3 && pair[1] === 8) relations.push({ dir: dir, stars: '三八', meaning: '三八为朋（木）— 助旺丁财' });
      if (pair[0] === 4 && pair[1] === 9) relations.push({ dir: dir, stars: '四九', meaning: '四九为友（金）— 喜庆、丁财' });
    });
    return relations;
  }

  function getXuankongAdvice(yunStar, wangShan, shanXiangJie) {
    var tips = [];
    var shenStar = NINE_STARS_JIXIONG[yunStar];

      tips.push('当元运星：' + shenStar.name + '（' + shenStar.wuxing + '，' + shenStar.jiXiong + '），' + shenStar.desc);
    }
    if (wangShan) {
      tips.push('卦得旺山旺向，丁财两旺，主事顺遂');
    } else {
      tips.push('目前非旺山旺向，须借重流年飞星催旺');
    }
    if (shanXiangJie.length > 0) {
      var good = shanXiangJie.filter(function(r) { return r.stars === '一六' || r.stars === '四九'; });
      var bad = shanXiangJie.filter(function(r) { return r.stars === '二七' || r.stars === '三八'; });
      if (good.length > 0) tips.push('山星向星会于' + good[0].dir + '：' + good[0].meaning);
      if (bad.length > 0) tips.push('山星向星会于' + bad[0].dir + '：' + bad[0].meaning + '，需化解');
    }
    return tips;
  }

  function checkMissingCorners(actualAreas) {
    if (!actualAreas || typeof actualAreas !== 'object') return [];
    var values = Object.values(actualAreas).filter(function(v) { return typeof v === 'number' && v > 0; });
    if (values.length === 0) return [];
    var sum = values.reduce(function(s, v) { return s + v; }, 0);
    var avg = sum / values.length;
    var missing = [];
    Object.keys(actualAreas).forEach(function(dir) {
      var ratio = actualAreas[dir] / avg;
      if (ratio < 0.7) {
        missing.push({
          direction: dir,
          ratio: ratio.toFixed(2),
          severity: ratio < 0.4 ? '严重' : ratio < 0.55 ? '中等' : '轻度',
          advice: _getMissingCornerAdvice(dir)
        });
      }
    });
    return missing;
  }

  function analyzeHuxing(huxingInfo) {

  function _computeHuxingScore(bazhaiTable, missing, center) {
    var score = 60;
    var good = _filterByRank(bazhaiTable, ['上吉','次吉','中吉']).length;
    score += (good - 4) * 3;
    missing.forEach(function(m) {
      score -= m.severity === '严重' ? 10 : m.severity === '中等' ? 5 : 2;
    });
    score += center.score;
    return Math.max(20, Math.min(100, score));
  }

  function _huxingAdvice(zhaiGua, missing, bazhaiTable) {
    var tips = [];
    if (missing.length > 0) {
      tips.push('宅有' + missing.length + '方缺角（' + missing.map(function(m){return m.direction;}).join('、') + '），需用对应五行物品补缺');
    }
    var shengQi = _filterByRank(bazhaiTable, ['上吉']);
    if (shengQi.length > 0) {
      tips.push('生气位在' + shengQi[0].dir + '方，宜作大门、主卧、书房');
    }
    var jueMing = _filterByRank(bazhaiTable, ['大凶']);
    if (jueMing.length > 0) {
      tips.push('绝命位在' + jueMing[0].dir + '方，宜作卫生间、储藏室，忌作卧室或入户门');
    }
    return tips;
  }

  function fengshuiAnalyze(opts) {
    opts = opts || {};
    var year = opts.year || 1990;
    var sex = opts.sex || 'male';
    var chaoxiang = opts.chaoxiang || '南';
    var buildYear = opts.buildYear || 2000;
    var huxingInfo = {
      huxing: opts.huxing || '三室一厅',
      chaoxiang: chaoxiang,
      louceng: opts.louceng || '6',
      area: opts.area || 100
    };
    var bazhai = bazhaiAnalyze(year, sex, chaoxiang);
    var xuankong = xuankongAnalyze(buildYear, chaoxiang);
    var huxing = analyzeHuxing(huxingInfo);
    var sanYao = null;
    if (opts.menhuDir && opts.zhuWoDir && opts.zaoFangDir) {
      sanYao = analyzeYangZhaiSanYao(year, sex, opts.menhuDir, opts.zhuWoDir, opts.zaoFangDir);
    }
    var totalScore = (bazhai.match ? 25 : 15) +
                     (xuankong.wangShanWangXiang ? 25 : 15) +
                     (huxing.score * 0.3);
    return {
      bazhai: bazhai,
      xuankong: xuankong,
      huxing: huxing,
      sanYao: sanYao,
      totalScore: Math.round(totalScore),
      summary: _summary({ bazhai: bazhai, xuankong: xuankong, huxing: huxing })
    };
  }

  function _summary(r) {
    var parts = [];
    if (r.bazhai.match) parts.push('宅命相配');
    else parts.push('宅命需化解');
    if (r.xuankong.wangShanWangXiang) parts.push('当元旺山旺向');
    else parts.push('非当运之宅');
    if (r.huxing.missing.length === 0) parts.push('户型方正');
    else parts.push('户型有' + r.huxing.missing.length + '方缺角');
    return parts.join('，');
  }
